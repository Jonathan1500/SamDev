// Nexdev · Multi-Gateway Payment Router
// Neonet + Epay + Stripe  ·  v4.1

namespace Nexdev\Payments;

readonly class GatewayRouter
{
  public function __construct(
    private array $drivers = [
      'neonet' => NeonetDriver::class,
      'epay'   => EpayDriver::class,
      'stripe' => StripeDriver::class,
    ]
  ) {}

  public function charge(
    string $gateway,
    Money  $amount,
    Customer $customer,
  ): PaymentResult {

    $driver = new $this->drivers[$gateway];
    $result = $driver->process($amount, $customer);

    if ($result->successful()) {
      Invoice::issue($result);
      Event::dispatch(new PaymentCompleted($result));
    }

    return $result;
  }
}
